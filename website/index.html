Drop-In Installation

Integrates seamlessly into any modern stack in less than 2 minutes.

import { HydeSecurityProvider } from '@tirth1107/hyde-security-js/react'

export default function App({ children }) {
  return (
    <HydeSecurityProvider
      config={{
        appName: 'My SaaS App',
        mode: 'strict',
        enableWatermark: true,
        onThreatDetected: (e) => console.log('Threat!', e)
      }}
    >
      {children}
    </HydeSecurityProvider>
  )
}
import { HydeSecurity } from '@tirth1107/hyde-security-js'

HydeSecurity.init({
  appName: 'My Vanilla App',
  mode: 'strict',
  enableWatermark: true,
  onThreatDetected: (e) => {
    fetch('/api/logs', { method: 'POST', body: JSON.stringify(e) })
  }
})

// Use Pro Features manually
HydeSecurity.modules.biometrics.startTracking()
HydeSecurity.modules.networkFirewall.enable(['api.myapp.com'])
